    **VIEW-03: Replace manual toggle with idle timer:**

    Replace `const [isAutoRotating, setIsAutoRotating] = useState(true)` with the idle timer pattern. Add these at the top of the component (after existing state declarations):

    ```ts
    // Idle timer for auto-rotate (VIEW-03)
    const [manualRotate, setManualRotate] = useState(true)  // user's explicit toggle choice
    const [idleActive, setIdleActive] = useState(false)     // whether user is currently idle
    const idleTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null)

    const startIdleTimer = useCallback(() => {
      if (idleTimerRef.current) clearTimeout(idleTimerRef.current)
      idleTimerRef.current = setTimeout(() => setIdleActive(true), 5000)
    }, [])

    const resetIdle = useCallback(() => {
      setIdleActive(false)
      if (manualRotate) startIdleTimer()
    }, [manualRotate, startIdleTimer])

    // Start idle timer on mount; clean up on unmount
    useEffect(() => {
      if (manualRotate) startIdleTimer()
      return () => {
        if (idleTimerRef.current) clearTimeout(idleTimerRef.current)
      }
    }, [manualRotate, startIdleTimer])
    ```

    Derived value: `const isAutoRotating = manualRotate && idleActive`

    Update the container div to reset idle on user interactions:
    ```tsx
    <div
      ref={containerRef}
      className="fixed inset-0 overflow-hidden"
      style={{ backgroundColor: '#0A0908' }}
      onMouseMove={resetIdle}
      onClick={resetIdle}
      onTouchStart={resetIdle}
      onKeyDown={resetIdle}
    >
    ```

    Update the auto-rotate toggle button (the RotateCw button in viewer controls):
    ```tsx
    <button
      onClick={() => {
        const next = !manualRotate
        setManualRotate(next)
        if (!next) {
          // User turned off — clear idle timer
          setIdleActive(false)
          if (idleTimerRef.current) clearTimeout(idleTimerRef.current)
        } else {
          // User turned on — restart idle timer
          startIdleTimer()
        }
      }}
      aria-label="Toggle auto-rotate"
      style={{ color: manualRotate ? '#D4A017' : '#A8A29E' }}
    >
      <RotateCw size={16} strokeWidth={1.5} />
    </button>
    ```

    Pass `autoRotate={isAutoRotating}` (the derived value) to PanoramaViewer (it already accepts this prop).

    **VIEW-02: F key keyboard shortcut:**

    Add a useEffect that registers/unregisters a document keydown listener:
    ```ts
    useEffect(() => {
      const handler = (e: KeyboardEvent) => {
        // Ignore if user is typing in an input
        if (e.target instanceof HTMLInputElement || e.target instanceof HTMLTextAreaElement) return
        if (e.key === 'f' || e.key === 'F') {
          e.preventDefault()
          toggleFullscreen()
        }
      }
      document.addEventListener('keydown', handler)
      return () => document.removeEventListener('keydown', handler)
    }, [toggleFullscreen])
    ```

    Also sync `isFullscreen` state with the browser's `fullscreenchange` event (so the button icon stays accurate if user exits fullscreen with Escape key):
    ```ts
    useEffect(() => {
      const handleFsChange = () => setIsFullscreen(!!document.fullscreenElement)
      document.addEventListener('fullscreenchange', handleFsChange)
      return () => document.removeEventListener('fullscreenchange', handleFsChange)
    }, [])
    ```
