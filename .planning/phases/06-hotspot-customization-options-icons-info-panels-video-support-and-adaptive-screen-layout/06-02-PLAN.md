    Write the file with these exact exports:

    ```typescript
    export type VideoType = 'youtube' | 'vimeo' | 'direct' | 'unknown'

    export interface ParsedVideo {
      type: VideoType
      embedSrc: string
    }

    export function parseVideoUrl(url: string): ParsedVideo {
      const isYoutube = /youtube\.com|youtu\.be/.test(url)
      const isVimeo = /vimeo\.com/.test(url)
      const isDirect = /\.(mp4|webm|ogg)(\?.*)?$/i.test(url)

      if (isYoutube) {
        const vidId = url.match(/(?:v=|youtu\.be\/)([^&?/]+)/)?.[1]
        return {
          type: 'youtube',
          embedSrc: vidId ? `https://www.youtube.com/embed/${vidId}` : url,
        }
      }

      if (isVimeo) {
        const vidId = url.match(/vimeo\.com\/(\d+)/)?.[1]
        return {
          type: 'vimeo',
          embedSrc: vidId ? `https://player.vimeo.com/video/${vidId}` : url,
        }
      }

      if (isDirect) {
        return { type: 'direct', embedSrc: url }
      }

      return { type: 'unknown', embedSrc: url }
    }
    ```

    No default export — named exports only (project convention). No 'use client' directive needed (pure utility function, no React).

    Create src/hooks/useViewerStore.ts. This store manages two pieces of ephemeral viewer state: which hotspot's info panel is open (activeHotspotId) and which video is shown in the full-screen modal (videoModalUrl). Both pieces of state span across the R3F Canvas boundary, which is why they need a Zustand store rather than component-local state or React context.

    Write the file exactly as:

    ```typescript
    'use client'

    import { create } from 'zustand'

    interface ViewerStore {
      // Active hotspot info panel state
      activeHotspotId: string | null
      setActiveHotspot: (id: string | null) => void
      // Full-screen video modal state
      videoModalUrl: string | null
      videoModalTitle: string | undefined
      openVideoModal: (url: string, title?: string) => void
      closeVideoModal: () => void
    }

    export const useViewerStore = create<ViewerStore>((set) => ({
      activeHotspotId: null,
      setActiveHotspot: (id) => set({ activeHotspotId: id }),
      videoModalUrl: null,
      videoModalTitle: undefined,
      openVideoModal: (url, title) => set({ videoModalUrl: url, videoModalTitle: title }),
      closeVideoModal: () => set({ videoModalUrl: null, videoModalTitle: undefined }),
    }))
    ```

    Pattern notes:
    - 'use client' directive is required — Zustand stores are used exclusively in client components
    - `import { create } from 'zustand'` — Zustand v5 named import (NOT `import create from 'zustand'` which is the v4 default export)
    - Named export `useViewerStore` (not default export per project convention)
    - activeHotspotId stores the Convex _id string of the clicked hotspot, or null when no panel is open
